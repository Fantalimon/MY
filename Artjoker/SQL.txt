    `id` int(10)  NOT NULL AUTO_INCREMENT PRIMARY KEY ,

    `email` varchar(150)  NOT NULL,
    `territory` varchar(250) COLLATE utf8_unicode_ci NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci ;
